设圆 $C:(x-a)^2+(y-b)^2=r^2,\text{定点}P(x_0,y_0)$
\begin{enumerate}[label=(\arabic*)]
    \item 点在圆上:\\
          $\overrightarrow{OP}=(x_0-a,y_0-b)\\
\vec{n}=(y_0-b,a-x_0)\\
\begin{aligned}[t]
    l:                                & (x_0-a)x+(y_0-b)y+C=0                     \\
    \text{带入}  (x_0,y_0) \text{得}: & C=ax_0+by_0-x_0^2+y_0^2                   \\
    l:                                & (x_0-a)x+(y_0-b)y+ax_0-x_0^2+by_0-y_0^2=0 \\
                                      & (x_0-a)x+(y_0-b)y+x_0(a-x_0)+y_0(b-y_0)=0 \\
                                      & (x_0-a)(x-x_0)+(y_0-b)(y-y_0)=0           \\
    \text{另:}C                       & = ax_0-x_0^2+by_0-y_0^2                   \\
                                      & =-(x_0-a)^2-(y_0-b)^2-ax_0-by_0+a^2+b^2   \\
                                      & =-r^2-a(x_0-a)-b(y_0-b)                   \\
    \text{带入C,得}:                  & (x_0-a)(x-a)+(y_0-b)(y-b)=r^2
\end{aligned}$\\
    法二：\\
    设切线方程为 $l:A(x-x_0)+B(y-y_0)=0 $(注意自由度为1)\\
    则圆心到 $l$ 距离:\\
$\begin{aligned}[t]
                  & \frac{|Aa-Ax_0+Bb-By_0|}{\sqrt{A^2+B^2}}=r \dots \circled{1}               \\
                  & (Aa-Ax_0+Bb-By_0)^2=(A^2+B^2)r^2                                           \\
    \text{带入r}: & A^2(x_0-a)^2+B^2(y_0-b)^2+2AB(x_0-a)(y_0-b)=(A^2+B^2)[(x_0-a)^2+(y_0-b)^2] \\
                  & B^2(a-x_0)^2+A^2(b-y_0)^2-2AB(x_0-a)(y_0-b)=0                              \\
                  & [B(x_0-a)-A(y_0-b)]^2=0                                                    \\
                  & B(x_0-a)-A(y_0-b)=0\dots \circled{2}                                       \\
                  & (x-x_0) \times \circled{2}+  (y_0-b)  \times  \circled{1}:                 \\
                  & (x_0-a)(x-x_0)+(y_0-b)(y-y_0)=0
\end{aligned}$\\
    也可用隐函数求导.
    \item 点在圆外:\\
$\begin{aligned}[t]
     & \frac{|Aa-Ax_0+Bb-By_0|}{\sqrt{A^2+B^2}}=r \dots \circled{1} \\
     & (Aa-Ax_0+Bb-By_0)^2=(A^2+B^2)r^2                             \\
     & A^2(x_0-a)^2+B^2(y_0-b)^2+2AB(x_0-a)(y_0-b)=(A^2+B^2)r^2     \\
     & B^2(a-x_0)^2+A^2(b-y_0)^2-2AB(x_0-a)(y_0-b)=0                \\
     & [B(x_0-a)-A(y_0-b)]^2=0                                      \\
     & B(x_0-a)-A(y_0-b)=0\dots \circled{2}                         \\
\end{aligned}$
\end{enumerate}
